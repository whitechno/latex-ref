    \path [name path=sloped line] (0,0) -- (30:1.5cm); 
    % (add `\usetikzlibrary{intersections}' after loading tikz in the preamble)
    \draw [name intersections={of=upward line and sloped line, by=x}] 
          [very thick,orange] (1,0) -- (x);
